    Import createAdminClient from '@/lib/supabase/admin'.
    Import type AnalyticsData, DauDataPoint, TopModeEntry from '@/types/analytics'.

    Export async function fetchAnalytics(): Promise<AnalyticsData>

    Body:
      const admin = createAdminClient()

      Step 1 — DAU for last 30 days:
      Supabase does not support raw GROUP BY in the JS client, so use .rpc() if a function
      exists, or fetch all user_progress rows updated in the last 30 days and group client-side.
      Use the client-side grouping approach (simpler, no new SQL function needed):

        const thirtyDaysAgo = new Date()
        thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30)
        const { data: progressRows } = await admin
          .from('user_progress')
          .select('user_id, updated_at')
          .gte('updated_at', thirtyDaysAgo.toISOString())

        Group by date (YYYY-MM-DD) and count distinct user_ids:
          const dauMap = new Map<string, Set<string>>()
          for (const row of progressRows ?? []) {
            const date = row.updated_at.slice(0, 10)
            if (!dauMap.has(date)) dauMap.set(date, new Set())
            dauMap.get(date)!.add(row.user_id)
          }
          Fill all 30 days (even zeros) by iterating from thirtyDaysAgo to today:
          const dau: DauDataPoint[] = []
          for (let i = 0; i < 30; i++) {
            const d = new Date(thirtyDaysAgo)
            d.setDate(d.getDate() + i)
            const dateStr = d.toISOString().slice(0, 10)
            dau.push({ date: dateStr, dau: dauMap.get(dateStr)?.size ?? 0 })
          }

      Step 2 — Top game modes from daily_completions:
      daily_completions has: user_id, challenge_date, elo_earned, completed_at
      This table tracks daily challenge completions, not per-mode completions.
      For top modes we fall back to a simplified approach: count daily_completions per
      challenge_date grouped by the associated daily_challenges.game_mode.
      Join the two tables:

        const { data: completionRows } = await admin
          .from('daily_completions')
          .select('challenge_date, daily_challenges!inner(game_mode)')
          .limit(1000)
          .order('challenge_date', { ascending: false })

        Group by game_mode and count:
          const modeMap = new Map<string, number>()
          for (const row of completionRows ?? []) {
            const mode = (row.daily_challenges as { game_mode: string } | null)?.game_mode ?? 'unknown'
            modeMap.set(mode, (modeMap.get(mode) ?? 0) + 1)
          }
          const topModes: TopModeEntry[] = Array.from(modeMap.entries())
            .map(([mode, completions]) => ({ mode, completions }))
            .sort((a, b) => b.completions - a.completions)
            .slice(0, 8)

        If topModes is empty (no completions data yet), return a placeholder:
          [{ mode: 'No data yet', completions: 0 }]

      Step 3 — Total counts:
        const [{ count: totalUsersCount }, { count: totalWordsCount }] = await Promise.all([
          admin.from('users').select('*', { count: 'exact', head: true }),
          admin.from('words').select('*', { count: 'exact', head: true }),
        ])

      Return: { dau, topModes, totalUsersCount: totalUsersCount ?? 0, totalWordsCount: totalWordsCount ?? 0 }

    Export interface DailyChallenge (inline, or import from types if preferred):
      id: string
      challenge_date: string    (ISO date string YYYY-MM-DD)
      game_mode: string
      vocabulary_ids: string[]  (array of word ID strings — stored as JSONB in DB)
      bonus_elo: number
      created_at: string

    Export async function listChallenges(): Promise<DailyChallenge[]>
      1. const admin = createAdminClient()
      2. const { data } = await admin
           .from('daily_challenges')
           .select('*')
           .order('challenge_date', { ascending: false })
           .limit(60)   (show last 60 entries — ~2 months)
      3. Return data ?? []

    Export async function upsertChallenge(formData: FormData): Promise<void>
      This is a Server Action (called directly as form action).
      1. Extract from formData:
           challengeDate = formData.get('challenge_date') as string   (YYYY-MM-DD)
           gameMode = formData.get('game_mode') as string
           vocabularyIdsRaw = formData.get('vocabulary_ids') as string
           bonusElo = parseInt(formData.get('bonus_elo') as string || '50', 10)
      2. Parse vocabulary_ids: split by comma, trim, filter empty:
           const vocabulary_ids = vocabularyIdsRaw
             .split(',')
             .map(s => s.trim())
             .filter(Boolean)
      3. Validate: if (!challengeDate || !gameMode) throw new Error('Date and game mode required')
      4. const admin = createAdminClient()
      5. UPSERT on challenge_date conflict:
           const { error } = await admin
             .from('daily_challenges')
             .upsert(
               { challenge_date: challengeDate, game_mode: gameMode, vocabulary_ids, bonus_elo: bonusElo },
               { onConflict: 'challenge_date' }
             )
      6. if (error) throw new Error('Failed to schedule challenge: ' + error.message)
      7. revalidatePath('/admin/daily-challenges')
