    Export async function listUsers(
      search: string = '',
      page: number = 1,
      pageSize: number = 50
    ): Promise<{ users: AdminUser[]; total: number }>
      1. const admin = createAdminClient()
      2. Query users table with left join to user_progress:
           const { data, error, count } = await admin
             .from('users')
             .select(`
               id,
               username,
               avatar_url,
               is_banned,
               is_admin,
               created_at,
               user_progress (
                 elo_score,
                 streak_count,
                 total_games,
                 last_practice_date
               )
             `, { count: 'exact' })
             .ilike('username', search ? '%' + search + '%' : '%')
             .order('created_at', { ascending: false })
             .range((page - 1) * pageSize, page * pageSize - 1)
      3. Map data to AdminUser shape:
           const users: AdminUser[] = (data ?? []).map(u => ({
             id: u.id,
             username: u.username,
             email: null,  // auth.users not accessible via public schema join
             avatar_url: u.avatar_url,
             is_banned: u.is_banned,
             is_admin: u.is_admin,
             created_at: u.created_at,
             elo_score: u.user_progress?.[0]?.elo_score ?? null,
             streak_count: u.user_progress?.[0]?.streak_count ?? null,
             total_games: u.user_progress?.[0]?.total_games ?? null,
             last_practice_date: u.user_progress?.[0]?.last_practice_date ?? null,
           }))
      4. Return { users, total: count ?? 0 }

    Note on the Supabase JS SDK nested select syntax: when joining a one-to-one relationship,
    the SDK may return the nested object as an array or object depending on cardinality detection.
    Use optional chaining with array index [0] as shown to safely handle both cases.

    Export async function banUser(userId: string): Promise<void>
      1. const admin = createAdminClient()
      2. const { error } = await admin
           .from('users')
           .update({ is_banned: true })
           .eq('id', userId)
      3. if (error) throw new Error('Failed to ban user: ' + error.message)
      4. revalidatePath('/admin/users')

    Export async function unbanUser(userId: string): Promise<void>
      1. const admin = createAdminClient()
      2. const { error } = await admin
           .from('users')
           .update({ is_banned: false })
           .eq('id', userId)
      3. if (error) throw new Error('Failed to unban user: ' + error.message)
      4. revalidatePath('/admin/users')
  </action>

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Admin Browser → Server Action | ban/unban actions carry user ID as bound parameter |
| Server Action → Supabase (service role) | UPDATE users SET is_banned — service role bypasses RLS |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-03-01 | Elevation of Privilege | banUser action | mitigate | is_admin check in middleware prevents non-admin from reaching /admin/users; banUser action itself does not re-verify caller, relying on middleware gate (defense in depth: acceptable for internal tool) |
| T-06-03-02 | Tampering | banUser / unbanUser userId parameter | mitigate | userId is bound server-side via .bind(null, user.id) — client cannot inject arbitrary UUIDs; UUID format validated implicitly by Supabase .eq() |
| T-06-03-03 | Denial of Service | Admin self-ban | mitigate | UI prevents ban button on is_admin=true rows (shows "(admin)" text instead); service role update not blocked at DB level — acceptable since UI gate is sufficient |
</threat_model>
