    Add 'use server' directive at top.
    Import createAdminClient from '@/lib/supabase/admin'.
    Import revalidatePath from 'next/cache'.
    Import type Word, WordFormData from '@/types/vocabulary'.

    Export async function listWords(
      category: string = 'all',
      page: number = 1,
      pageSize: number = 20
    ): Promise<{ words: Word[]; total: number }>
      1. const admin = createAdminClient()
      2. Build query: admin.from('words').select('*', { count: 'exact' })
      3. If category !== 'all': .eq('category', category)
      4. Apply ordering: .order('created_at', { ascending: false })
      5. Apply pagination: .range((page - 1) * pageSize, page * pageSize - 1)
      6. Return { words: data ?? [], total: count ?? 0 }

    Export async function listCategories(): Promise<string[]>
      1. const admin = createAdminClient()
      2. const { data } = await admin.from('words').select('category').order('category')
      3. Return Array.from(new Set((data ?? []).map(r => r.category))).filter(Boolean)

    Export async function createWord(formData: FormData): Promise<void>
      This is called from form action={createWord}.
      1. Extract fields from FormData:
           word = formData.get('word') as string
           classifier = formData.get('classifier') as string (default '')
           category = formData.get('category') as string (default 'general')
           distractorsRaw = formData.get('distractors') as string
           distractors = distractorsRaw.split(',').map(s => s.trim()).filter(Boolean)
           imageFile = formData.get('image') as File | null
      2. const admin = createAdminClient()
      3. Upload image if imageFile exists and imageFile.size > 0:
           const fileBuffer = Buffer.from(await imageFile.arrayBuffer())
           const fileName = Date.now() + '-' + imageFile.name.replace(/[^a-zA-Z0-9._-]/g, '_')
           const { data: uploadData, error: uploadError } = await admin.storage
             .from('vocabulary-images')
             .upload(fileName, fileBuffer, { contentType: imageFile.type, upsert: false })
           if (uploadError) throw new Error('Image upload failed: ' + uploadError.message)
           const { data: urlData } = admin.storage.from('vocabulary-images').getPublicUrl(uploadData.path)
           imageUrl = urlData.publicUrl
      4. Insert word row:
           const { error } = await admin.from('words').insert({
             word, classifier, category, image_url: imageUrl ?? null, distractors
           })
           if (error) throw new Error(error.message)
      5. revalidatePath('/admin/vocabulary')
      (redirect is handled by the calling page via useFormState or after the action)

    Export async function updateWord(id: number, formData: FormData): Promise<void>
      Same extraction logic as createWord.
      If no new imageFile provided (imageFile.size === 0), keep existing image_url (read it from
      a hidden formData field named 'existing_image_url').
      Call admin.from('words').update({ word, classifier, category, image_url, distractors }).eq('id', id)
      revalidatePath('/admin/vocabulary')
      revalidatePath('/admin/vocabulary/' + id + '/edit')

    Export async function deleteWord(id: number): Promise<void>
      1. const admin = createAdminClient()
      2. const { data: word } = await admin.from('words').select('image_url').eq('id', id).single()
      3. If word?.image_url: extract path from URL and delete from storage:
           const path = word.image_url.split('/vocabulary-images/')[1]
           if (path) await admin.storage.from('vocabulary-images').remove([path])
      4. await admin.from('words').delete().eq('id', id)
      5. revalidatePath('/admin/vocabulary')

    IMPORTANT: updateWord and deleteWord accept numeric id as first argument. The calling page
    passes id via a hidden form field and uses bind() or a wrapper:
      const updateWordWithId = updateWord.bind(null, wordId)

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Admin Browser → Server Action | Form data including file upload crosses trust boundary |
| Server Action → Supabase Storage | File bytes sent to vocabulary-images bucket via service role |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-02-01 | Tampering | createWord Server Action | mitigate | Server Action runs server-side only; FormData values are strings (no eval); distractors split by comma, no SQL injection possible via Supabase JS client parameterized queries |
| T-06-02-02 | Information Disclosure | Image upload | mitigate | File uploaded via service role client (server-side); public URL exposed is intentional (vocabulary images are public); no user data in filenames (use timestamp + sanitized name) |
| T-06-02-03 | Denial of Service | Unrestricted file size upload | accept | Admin-only dashboard; file size limits enforced by Supabase Storage (default 50MB); acceptable risk for internal tool |
| T-06-02-04 | Tampering | deleteWord removes storage file | mitigate | Path extracted from URL string before deletion; URL must contain '/vocabulary-images/' segment — malformed URLs skip storage delete gracefully |
</threat_model>
