    --- ShimmerBox ---
    @Composable
    fun ShimmerBox(
        modifier: Modifier = Modifier,
        shape: Shape = RoundedCornerShape(8.dp)
    )
    Implementation (no external library — use InfiniteTransition only):
      val infiniteTransition = rememberInfiniteTransition(label = "shimmer")
      val alpha by infiniteTransition.animateFloat(
          initialValue = 0.3f,
          targetValue = 0.9f,
          animationSpec = infiniteRepeatable(
              animation = tween(durationMillis = 800, easing = LinearEasing),
              repeatMode = RepeatMode.Reverse
          ),
          label = "shimmer_alpha"
      )
      Box(modifier = modifier.clip(shape).background(MaterialTheme.colorScheme.surfaceVariant.copy(alpha = alpha)))

    Usage note in file kdoc: "Use ShimmerBox as a placeholder with the same size as the real content:
    ShimmerBox(Modifier.fillMaxWidth().height(120.dp)) while data loads, then replace with real content."
