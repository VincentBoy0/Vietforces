    Step B — Create RemoteProgressSource.kt at package com.example.vietforces.data.remote:
    - Imports: io.github.jan.supabase.SupabaseClient, io.github.jan.supabase.postgrest.from,
      io.github.jan.supabase.postgrest.query.Columns, javax.inject.Inject, javax.inject.Singleton
    - @Singleton class RemoteProgressSource @Inject constructor(private val supabase: SupabaseClient)
    - suspend fun upsertProgress(dto: UserProgressDto): Result<Unit>:
        try {
            supabase.from("user_progress").upsert(dto) { onConflict = "user_id" }
            Result.success(Unit)
        } catch (e: Exception) { Result.failure(e) }
    - suspend fun fetchProgress(userId: String): Result<UserProgressDto?>:
        try {
            val result = supabase.from("user_progress")
                .select()
                .eq("user_id", userId)
                .decodeSingleOrNull<UserProgressDto>()
            Result.success(result)
        } catch (e: Exception) { Result.failure(e) }

    Step C — Create ProgressRepository.kt at package com.example.vietforces.data.repository:
    - Imports: com.example.vietforces.data.manager.UserProgressManager,
      com.example.vietforces.data.remote.RemoteProgressSource,
      com.example.vietforces.data.model.UserProgressDto,
      javax.inject.Inject, javax.inject.Singleton, java.time.Instant
    - @Singleton class ProgressRepository @Inject constructor(
          private val remoteSource: RemoteProgressSource,
          private val authRepository: AuthRepository
      )
    - private fun buildDtoFromLocal(userId: String): UserProgressDto:
        val session = UserProgressManager.getUserSession()
        UserProgressDto(
            userId = userId,
            eloRating = session.eloRating,
            currentStreak = session.currentStreak,
            longestStreak = session.longestStreak,
            wordsLearnedCount = session.learnedWordIds.size,
            lastPracticed = session.lastPracticeDate,
            updatedAt = Instant.now().toString()
        )
    - suspend fun syncToCloud(): Result<Unit>:
        val userId = authRepository.currentUserId ?: return Result.failure(Exception("Not logged in"))
        remoteSource.upsertProgress(buildDtoFromLocal(userId))
    - suspend fun loadFromCloud(): Result<Unit>:
        val userId = authRepository.currentUserId ?: return Result.failure(Exception("Not logged in"))
        val result = remoteSource.fetchProgress(userId).getOrNull() ?: return Result.success(Unit)
        // Last-write-wins: overwrite local only if cloud is newer
        val localDate = UserProgressManager.getUserSession().lastPracticeDate
        val cloudDate = result.lastPracticed
        if (cloudDate.isNotEmpty() && (localDate.isEmpty() || cloudDate > localDate)) {
            val session = UserProgressManager.getUserSession()
            session.eloRating = result.eloRating
            session.currentStreak = result.currentStreak
            session.longestStreak = result.longestStreak
            session.lastPracticeDate = result.lastPracticed
            // Note: learnedWordIds count cannot be restored from count alone — keep local set,
            // but record the cloud count for UI display. Actual word IDs stay local.
        }
        Result.success(Unit)
    - suspend fun syncIfLoggedIn(): Result<Unit>:
        if (authRepository.currentUserId == null) return Result.success(Unit)
        return syncToCloud()

    Note on UserSession mutability: UserProgressManager.getUserSession() returns a mutable UserSession
    object (not a copy). Direct field mutation is intentional here to match the existing codebase pattern
    (see UserProgressManager.recordCorrectAnswer which also mutates userSession directly).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| ProgressRepository → Supabase postgrest | Progress data (ELO, streak) sent to cloud over TLS |
| Supabase RLS → user_progress table | Row-level security must restrict each user to their own row |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-09 | Tampering | user_progress upsert | mitigate | Supabase RLS policy on user_progress must enforce: auth.uid() = user_id. The client sends user_id from auth token, not from local state. Note: verify in Supabase dashboard that the RLS policy was created in Phase 1 (FOUND-01). If absent, add: CREATE POLICY "Users can upsert own progress" ON user_progress FOR ALL USING (auth.uid() = user_id). |
| T-02-10 | Information Disclosure | ProgressRepository.loadFromCloud | accept | Cloud progress (ELO, streak) is non-sensitive; it is the same data displayed publicly on the leaderboard |
| T-02-11 | Denial of Service | Foreground sync in onResume | accept | Sync is a lightweight upsert; a malicious user calling onResume rapidly would hit Supabase rate limits (handled server-side) |
| T-02-12 | Tampering | MigrationService — migration_completed flag | accept | Flag gates re-migration, not security; a cleared flag just re-migrates the same local data (idempotent) |
| T-02-SC | Tampering | No new package installs | accept | No new dependencies; uses existing supabase-postgrest-kt already in build.gradle |
</threat_model>
