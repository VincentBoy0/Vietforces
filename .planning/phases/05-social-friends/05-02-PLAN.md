    ```kotlin
    suspend fun getFriendsLeaderboard(): Result<List<LeaderboardEntry>> {
        return try {
            val userId = authRepository.currentUserId
                ?: return Result.success(emptyList())

            // Step 1: get the IDs the current user follows
            @Serializable data class FollowRow(@SerialName("following_id") val followingId: String)
            val followingIds = supabase.from("friendships")
                .select("following_id") {
                    filter { eq("follower_id", userId) }
                }
                .decodeList<FollowRow>()
                .map { it.followingId }

            if (followingIds.isEmpty()) return Result.success(emptyList())

            // Step 2: query leaderboard for those IDs
            val entries = supabase.from("leaderboard")
                .select {
                    filter { isIn("user_id", followingIds) }
                    order("elo_score", Order.DESCENDING)
                }
                .decodeList<LeaderboardEntry>()

            Result.success(entries)
        } catch (e: Exception) {
            Result.failure(e)
        }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → users SELECT (search) | Permissive SELECT granted by migration 007; only id + username projected in query |
| client → friendships INSERT (follow) | RLS WITH CHECK (follower_id = auth.uid()) prevents following as another user |
| client → friendships DELETE (unfollow) | RLS USING (follower_id = auth.uid()) prevents deleting others' follows |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-02-01 | Spoofing | followUser — follower_id injected | mitigate | RLS `WITH CHECK (follower_id = auth.uid())` enforced server-side; client cannot set arbitrary follower_id |
| T-05-02-02 | Information Disclosure | searchUsers ILIKE returns all users | accept | Only username + id returned; no PII (email, fcm_token excluded by projected columns) |
| T-05-02-03 | Denial of Service | unbounded search results | mitigate | limit(20L) on all searchUsers queries |
| T-05-02-04 | Denial of Service | debounce bypass (rapid typing) | mitigate | searchJob?.cancel() + delay(300L) ensures at most one in-flight search per session |
| T-05-02-05 | Tampering | optimistic follow update reverted incorrectly | mitigate | on failure: revert to previous followingIds set and surface error in uiState.error |
</threat_model>
