    streak_freeze_count integer NOT NULL DEFAULT 0,
    last_practice_date date, total_games integer NOT NULL DEFAULT 0,
    words_learned jsonb NOT NULL DEFAULT '[]'::jsonb,
    updated_at timestamptz NOT NULL DEFAULT now().
    UNIQUE constraint on user_id.
    RLS: ENABLE ROW LEVEL SECURITY.
    Policy "progress_select_own": SELECT USING (user_id = auth.uid()).
    Policy "progress_insert_own": INSERT WITH CHECK (user_id = auth.uid()).
    Policy "progress_update_own": UPDATE USING (user_id = auth.uid()) WITH CHECK (user_id = auth.uid()).

    --- TABLE: public.leaderboard ---
    Columns: user_id uuid PRIMARY KEY REFERENCES public.users(id) ON DELETE CASCADE,
    username text NOT NULL, elo_score integer NOT NULL DEFAULT 0,
    weekly_elo integer NOT NULL DEFAULT 0,
    rank_tier text NOT NULL DEFAULT 'Bronze',
    updated_at timestamptz NOT NULL DEFAULT now().
    RLS: ENABLE ROW LEVEL SECURITY.
    Policy "leaderboard_select_all": SELECT USING (true)  — public read.
    Policy "leaderboard_service_write": INSERT and UPDATE restricted to service_role using
    TO service_role clause (no USING/WITH CHECK needed when scoped to role).

    --- TABLE: public.daily_challenges ---
    Columns: id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    challenge_date date UNIQUE NOT NULL,
    game_mode text NOT NULL,
    vocabulary_ids jsonb NOT NULL DEFAULT '[]'::jsonb,
    bonus_elo integer NOT NULL DEFAULT 50,
    created_at timestamptz NOT NULL DEFAULT now().
    RLS: ENABLE ROW LEVEL SECURITY.
    Policy "challenges_select_all": SELECT USING (true)  — public read.
    Policy "challenges_service_insert": INSERT TO service_role (no WITH CHECK needed).

    --- TABLE: public.friendships ---
    Columns: follower_id uuid NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| authenticated user → PostgREST | All data access goes through RLS; auth.uid() binds row access to identity |
| service_role → leaderboard / daily_challenges | Only server-side code (Edge Functions, pg_cron) may write these tables |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Elevation of Privilege | user_progress writable by others | mitigate | RLS policy "progress_insert_own" uses WITH CHECK (user_id = auth.uid()) — prevents inserting rows for other users |
| T-02-02 | Tampering | leaderboard ELO writable by client | mitigate | leaderboard INSERT/UPDATE policies are TO service_role — anon/authenticated clients cannot write; ELO updates go through server functions (Phase 3) |
| T-02-03 | Information Disclosure | users table exposes all rows | mitigate | "users_select_own" policy limits SELECT to own row; other users' profiles are not readable at this phase (Phase 5 social adds explicit public profile policy) |
| T-02-04 | Spoofing | auth.uid() manipulation | accept | auth.uid() is derived from the verified JWT by PostgREST/Supabase; client cannot forge it |
| T-02-SC | Tampering | SQL migration applied to wrong project | accept | Migration is idempotent; reviewed before applying; no destructive DROP statements |
</threat_model>
