      workflow_dispatch: (empty — allows manual triggering from GitHub Actions UI for testing)

    jobs block with a single job named keepalive:
      runs-on: ubuntu-latest
      steps:
        - name: Ping Supabase REST API
          Run a curl command that:
          - GETs the Supabase REST health endpoint: ${{ secrets.SUPABASE_URL }}/rest/v1/
          - Includes the apikey header: -H "apikey: ${{ secrets.SUPABASE_ANON_KEY }}"
          - Includes Accept: application/json header
          - Uses --fail-with-body flag so the step fails (non-zero exit) on HTTP 4xx/5xx
          - Uses --silent --show-error flags to suppress progress but show errors
          - Uses --max-time 30 to timeout after 30 seconds
          The curl command must be a single run: | block.

        - name: Log result
          Run echo "Supabase keepalive ping successful at $(date -u)" to record the
          timestamp in the Actions log. Add if: success() condition so it only runs on
          success.

    The workflow YAML must use ${{ secrets.SUPABASE_URL }} and ${{ secrets.SUPABASE_ANON_KEY }}
    — never hardcode URLs or keys in the file. These secrets must be added to the GitHub repo
    settings by the developer (documented in user_setup frontmatter above).

    Add a comment at the top of the file explaining: "Pings the Supabase REST API daily to
    prevent the free-tier project from pausing after 7 days of inactivity. Requires
