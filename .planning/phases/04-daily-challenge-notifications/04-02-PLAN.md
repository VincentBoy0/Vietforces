Public methods:

1. suspend fun getTodayChallenge(): Result<DailyChallenge?>

   - Wrap in try/catch, return Result.success(value) or Result.failure(e)

2. suspend fun getTodayCompletion(): Result<DailyCompletion?>
   - Return Result.failure if authRepository.currentUserId is null
   - Query daily_completions with filter: eq("user_id", userId) AND eq("challenge_date", todayUtcString())
   - Return Result.success(decodeSingleOrNull)

3. suspend fun submitCompletion(challengeDate: String): Result<Int>
   - Return Result.failure if not logged in
   - Call supabase.postgrest.rpc("award_daily_bonus", buildJsonObject { put("p_user_id", userId); put("p_challenge_date", challengeDate) }).decodeAs<Int>()
   - On success: also call StreakRepository.instance?.updateStreak() (fire-and-forget result)

4. suspend fun getLast7DaysHistory(): Result<List<DailyChallengeHistoryItem>>
   - Compute cutoff = 6 days ago UTC (use Calendar.getInstance(TimeZone.UTC) - 6 days)
   - Query daily_challenges WHERE challenge_date >= cutoff ORDER BY challenge_date ASC → List<DailyChallenge>
   - If logged in: query daily_completions WHERE user_id = userId AND challenge_date >= cutoff → Set<String> of completed dates
   - Map each DailyChallenge to DailyChallengeHistoryItem: isCompleted = challengeDate in completedDates, eloEarned = completion.eloEarned or 0
   - Return Result.success(list)

Package: com.example.vietforces.ui.viewmodel

Sealed class DailyChallengeUiState:
  - Loading
  - NoChallenge (no challenge for today)
  - Ready(challenge: DailyChallenge, isCompleted: Boolean, eloEarned: Int, history: List<DailyChallengeHistoryItem>, countdownSeconds: Long)
  - Completed(eloEarned: Int, streakUpdated: Boolean)
  - Error(message: String)

@HiltViewModel class DailyChallengeViewModel @Inject constructor(private val repository: DailyChallengeRepository) : ViewModel()

Fields:
  - private val _uiState = MutableStateFlow<DailyChallengeUiState>(DailyChallengeUiState.Loading)
  - val uiState: StateFlow<DailyChallengeUiState> = _uiState.asStateFlow()
  - private var countdownJob: Job? = null

init block: call loadChallenge()

fun loadChallenge(): launch in viewModelScope:
  1. _uiState.value = Loading
  2. val challengeResult = repository.getTodayChallenge()
  3. If failure or null → _uiState.value = NoChallenge; return
  4. val challenge = challengeResult.getOrNull() ?: return (emit NoChallenge)
  5. val completionResult = repository.getTodayCompletion()
  6. val completion = completionResult.getOrNull()
  7. val historyResult = repository.getLast7DaysHistory()
  8. val history = historyResult.getOrDefault(emptyList())
  9. _uiState.value = Ready(challenge, isCompleted = completion != null, eloEarned = completion?.eloEarned ?: 0, history = history, countdownSeconds = computeSecondsUntilMidnightUtc())
  10. If not completed: startCountdown()

fun computeSecondsUntilMidnightUtc(): Long — compute milliseconds until next 00:00:00 UTC using Calendar.getInstance(TimeZone.getTimeZone("UTC")): set time to tomorrow midnight, subtract System.currentTimeMillis(), divide by 1000.

fun startCountdown(): cancel existing countdownJob first; countdownJob = viewModelScope.launch { while (true) { delay(1000L); val state = _uiState.value; if (state is Ready) { val secs = computeSecondsUntilMidnightUtc(); if (secs <= 0L) { loadChallenge(); break }; _uiState.value = state.copy(countdownSeconds = secs) } else break } }

fun submitCompletion(challengeDate: String): launch in viewModelScope:
  1. val result = repository.submitCompletion(challengeDate)
  2. result.onSuccess { eloEarned -> _uiState.value = Completed(eloEarned = eloEarned, streakUpdated = true); countdownJob?.cancel() }
  3. result.onFailure { e -> if (e.message?.contains("already_completed") == true) { loadChallenge() } else { _uiState.value = Error(e.message ?: "Lỗi khi hoàn thành thử thách") } }

override fun onCleared(): countdownJob?.cancel(); super.onCleared()
